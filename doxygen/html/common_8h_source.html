220  inline int wordCount(const std::string &str) {
221  int c = 0;
222  std::string::const_iterator pos = str.begin();
223  std::string::const_iterator end = str.end();
224 
225  while(pos != end) {

241  inline std::string getStdoutFromCommand(std::string cmd) {
242  try {
243  std::string data;
244  std::FILE *stream;
245  int MAX_BUFFER = 65535;
246  char buffer[MAX_BUFFER];
247  cmd.append(" 2>&1");

250  if (!stream){

287  void reset(const std::string& src, const std::string& delim) {
288  std::vector<std::string> tokens;
289  std::string::size_type start = 0;
290  std::string::size_type end;
291 
292  for ( ; ; ) {
293  end = src.find(delim, start);
294  tokens.push_back(src.substr(start, end - start));
295 
296  if (end == std::string::npos)
297  break;
298 
299  start = end + delim.size();
300  }
301 
302  _tokens.swap (tokens);
303  }
